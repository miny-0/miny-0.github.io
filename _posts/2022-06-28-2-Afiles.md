#include<stdio.h>

void main() {
	float a, b, c, ans;
	ans = 0;
	char op, g;



	while (1) {
		printf("o를 누르면 연산, n누르면 마침");
		scanf_s("%c", &g);

		if (g == 'o') {
			printf("연산할 숫자를 입력하시오. ");
			scanf_s("%f", &a);
			getchar();


			printf("연산할 숫자를 입력하시오.");
			scanf_s("%f", &b);
			getchar();

			printf("연산자를 입력하시오. 끝내려면 x를 누르시오.");
			scanf_s("%c", &op);
			getchar();

			switch (op)
			{
			case '+':
				ans = a + b;
				printf("%f\n", ans);
				break;

			case '-':
				ans = a - b;
				printf("%f\n", ans);
				break;

			case '*':
				ans = a * b;
				printf("%f\n", ans);
				break;
			case '/':
				ans = a / b;
				printf("%f\n", ans);
				break;

			case 'x':
				return 1;
				break;

			default:
				0;
				break;
			}
		}
		else if (g == 'n') {
			return 1;
		}
	}
}
