## 첫번째 코딩테스트C
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

long long solution(int a, int b) {
    int i;
    long long answer = 0;

    if(a>b){
        for(i=b;i<=a;i++){
            answer += i;

        }
        return answer;
    }
    else if(b>a){
        for(i=a;i<=b;i++){
            answer += i;

        }
        return answer;
    }

    else {
        answer = b;
        return answer;
    }
}
